###  原理及操作

```C
int lowbit(int x)
{
    return x & -x;
}

void add(int x, int k)//表示将序列中第x个数加上k。
{
    for(int i = x; i <= n; i += lowbit(i))
        t[i] += k;
}
//ask(x)表示将查询序列前x个数的和
int ask(int x)
{
    int sum = 0;
    for(int i = x; i; i -= lowbit(i))
        sum += t[i];
    return sum;
}
```
